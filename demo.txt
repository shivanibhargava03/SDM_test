class SelectionSort(){
    
    void bsort(int arr[]){
      int n=arr.length();
      for(int i=0;i<n-i;i++){
        boolean flag =false;
        for(int j=0;j<n-i-1;j++){
            if(arr[j]>arr[j+1]){
                flag=true;
                int temp=arr[j+1];
                arr[j+1]=arr[j];
                arr[j]=temp;
            }

            if(!flag){
                return;
            }
        }
    }
}
